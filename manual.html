Step-1. Paste Pine Script into TradingView

Main Pine Script

//@version=5
indicator("SEND TO DB", overlay=true)

var int streak = 0
var bool streakBullish = na
var float streakOpen = na
var int oppositeCount = 0
var bool waitingForConfirmation = false
var bool alertTriggered = false

// Candle direction
isGreen = close > open
isRed = close < open
thisBullish = isGreen

// Reset alert each bar
alertTriggered := false

if not waitingForConfirmation
    if streak == 0
        streak := 1
        streakBullish := thisBullish
        streakOpen := open
    else if thisBullish == streakBullish
        streak += 1
        streakOpen := open
    else
        if streak >= 4
            waitingForConfirmation := true
            oppositeCount := 1
            if (streakBullish and low <= streakOpen) or (not streakBullish and high >= streakOpen)
                streak := 1
                streakBullish := thisBullish
                streakOpen := open
                waitingForConfirmation := false
                oppositeCount := 0
        else
            streak := 1
            streakBullish := thisBullish
            streakOpen := open
else
    if thisBullish != streakBullish
        oppositeCount += 1
        if oppositeCount > 2
            streak := 1
            streakBullish := thisBullish
            waitingForConfirmation := false
            oppositeCount := 0
            streakOpen := open
    else
        if oppositeCount == 2
            if (streakBullish and low <= streakOpen) or (not streakBullish and high >= streakOpen)
                streak := 1
                streakBullish := thisBullish
                waitingForConfirmation := false
                oppositeCount := 0
                streakOpen := open
            else
                alertTriggered := true
                streak := 1
                streakBullish := thisBullish
                waitingForConfirmation := false
                oppositeCount := 0
                streakOpen := open
        else if oppositeCount == 1
            alertTriggered := true
            streak := 1
            streakBullish := thisBullish
            waitingForConfirmation := false
            oppositeCount := 0
            streakOpen := open

// Plot shape on chart
plotshape(alertTriggered, title="Pattern Alert", location=location.abovebar, style=shape.labelup, color=color.orange, size=size.small, text="ALERT")

json_payload = '{"ticker":"' + syminfo.ticker + '","time":"' + str.tostring(time, "yyyy-MM-dd HH:mm") + '","ohlc":[' +

Economic News Automation (GitHub Cron)

This system uses GitHub Actions as a free VPS. News is fetched automatically every 5 minutes and sent to your database.

How it Works

  • GitHub runs a cron job every 5 minutes (UTC)
  • Python scraper fetches economic calendar data
  • Parsed events are sent to insert_event.php
  • No browser, no Google Script, no manual trigger

1️⃣ GitHub Workflow (calendar-cron.yml)

name: Economic Calendar Cron

on:
  schedule:
    - cron: "30 3 * * *"    # 9:00 AM IST
    - cron: "0 8 * * *"     # 1:30 PM IST
    - cron: "0 12 * * *"    # 5:30 PM IST
    - cron: "0 15 * * *"    # 8:30 PM IST
  workflow_dispatch:

jobs:
  run-calendar:
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Install system dependencies
        run: |
          sudo apt-get update
          sudo apt-get install -y \
            libatk-bridge2.0-0 \
            libatk1.0-0 \
            libcups2 \
            libxcomposite1 \
            libxdamage1 \
            libxrandr2 \
            libgbm1 \
            libasound2t64 \
            libpangocairo-1.0-0 \
            libxkbcommon0

      - name: Install Python dependencies
        run: |
          pip install -r requirements.txt
          playwright install chromium

      - name: Run scraper (single execution)
        run: |
          echo "⏱ Running scraper once"
          python scraper.py

2️⃣ Python Requirements (requirements.txt)

requests
pytz
playwright
⚠ Playwright is used only if the site blocks normal requests.

3️⃣ Python Scraper (scraper.py)

import requests
import pytz
from datetime import datetime

INSERT_URL = "https://your.site/insert_event.php"

def send_event(event_name, impact, event_time):
    payload = {
        "event_name": event_name,
        "impact": impact,
        "event_time": event_time,
        "sent_status": 0
    }
    requests.post(INSERT_URL, data=payload, timeout=15)

def main():
    # Example data (replace with real parser logic)
    now_ist = datetime.now(pytz.timezone("Asia/Kolkata"))
    event_time = now_ist.strftime("%Y-%m-%d %H:%M:%S")

    send_event(
        event_name="[USD] Sample Economic Event",
        impact=1,
        event_time=event_time
    )

if __name__ == "__main__":
    main()

4️⃣ Timezone Info

GitHub cron runs in UTC. All event times are converted to IST (GMT+5:30) inside Python.

5️⃣ Run Frequency

*/5 * * * *   → every 5 minutes
GitHub does NOT allow 1-minute cron jobs.
Minimum is 5 minutes (this is a GitHub limitation).

6️⃣ Manual Run (Testing)

Go to:
GitHub → Actions → Economic Calendar Cron → Run workflow
This lets you test instantly without waiting.
